select session_id, CAST(VARCHAR_FORMAT(start_time, 'YYYY-MM-DD HH24:MI:SS') as char(19)) as START_TIME, timestampdiff(2, current_timestamp-start_time) as ELAPSED_SEC, cast(client_name as char(20)) as CLIENT_NAME, cast(state as char(6)) as STATE, cast(bytes_sent as decimal(18,0)) as BYTES_SENT, cast(bytes_received as decimal(18,0)) as BYTES_RECEIVED, cast(cast(bytes_received as decimal(18,0))/1024/1024/timestampdiff(2, current_timestamp-start_time) as decimal(10,2)) as RECV_MBPS from sessions where client_name like '$1%'
